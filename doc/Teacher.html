def find_by_id(id)
      file  = File.new('teachers.csv', 'r')
  file.each_line("\n") do |row|
    info = row.split(", ")
    if info[0].to_i == id
      @@id = info[0].to_i
      @@name = info[1]
      @@surname = info[2]
      @discipline = info[3]
      return self
    end
  end 
end

def show_all
  file = File.new('teachers.csv', 'r')
  file.each_line("\n") do |row|
    teacher = Teacher.new(row.split(", "))
    teacher.show
  end         
end
